		\subsubsection{Zona}

			\paragraph{Especificación de atributos: \\}
			
			\begin{itemize}
				\item{\emph{ID\_Zona:}} Identificador de la zona.
			\end{itemize}
			
			\paragraph*{Especificación de identificador único:}
			
			El identificador único es \emph{ID\_Zona}.

		\subsubsection{Zona PC}
			\paragraph{Definición:}

			\paragraph{Especificación de atributos: \\}
			
			\begin{itemize}
				\item{\emph{ID\_Zona:}} Identificador de la zona dentro de todas las zonas de PCs.
			\end{itemize}
			
			\paragraph*{Especificación de identificador único:}

		\subsubsection{Zona Componentes}
			\paragraph{Definición:}

			\paragraph{Especificación de atributos: \\}
			
			\begin{itemize}

		\subsubsection{Hueco Componente}
			\paragraph{Definición: } 
				
			Lugar físico donde se almacena un único componente. Cada Zona se divide en huecos que cada uno tiene asignada una Columna y una Altura.
			
			\paragraph{Especificación de atributos: \\}
			
			\begin{itemize}
				\item{\emph{Columna:}} Indicador de la columna dentro de la zona.
				\item{\emph{Altura:}} Indicador de la altura dentro de la zona.
			\end{itemize}
			
			\paragraph*{Especificación de identificador único:}
			
			El identificador único es \emph{ID\_Zona}+\emph{Columna}+\emph{Altura}.
			
		\subsubsection{Componente}
			\paragraph{Definición:} 
			
			Elemento utilizado en el armado de una PC.
			
			\paragraph{Especificación de atributos: \\}
			
			\begin{itemize}
				\item{\emph{Número de Serie:}} Número que identifica univocamente a un componente.
				\item{\emph{Estado:}} Indica si el componente ya se encuentra en manos de la empresa (Recibido) o si ha sido pedido pero todavía no se encuentra a disposición(Pedido). Este atributo es calculable.
			
				\item{\emph{Fecha de llegada:}} Fecha en la que el componente arribó al almacén.
				\item{\emph{Descripción:}} Características partículares del componente.
			\end{itemize}
			\paragraph*{Especificación de identificador único:}
			
			El identificador único es \emph{Número de Serie}.

		\subsubsection{Suministro}
			\paragraph{Definición:} 
			
			Representa a un pedido realizado al proveedor.
			
			
			
			\paragraph{Especificación de atributos: \\}
			
			\begin{itemize}
				\item{\emph{Número Suministro:}} Número que identifica univocamente a un suministro.
				\item{\emph{Emitido:}} Indica si el pedido de suministro ya fue impreso.
			\end{itemize}
			
			\paragraph*{Especificación de identificador único:}
			
			El identificador único es \emph{Número Suministro}.

		\subsubsection{Proveedor}
			\paragraph{Definición:} 
			
			Abastecedor de Componentes usados para la fabricación de PCs.
			
			\paragraph{Especificación de atributos: \\}
			
			\begin{itemize}
				\item{\emph{CUIT:}} Clave única de indentificación tributaria que sirve para poder identificar inequívocamente a las personas físicas o jurídicas.
				\item{\emph{Nombre:}} Nombre o razon social de la empresa.
				\item{\emph{FAX:}} Número de teléfono en donde el proveedor puede recibir un FAX.
				\item{\emph{Dirección:}} Dirección legal, en donde reside el proveedor.
				\item{\emph{Teléfono:}} Número de teléfono del proveedor.
				\item{\emph{Mail:}} E-mail del proveedor.
			\end{itemize}
			
			\paragraph*{Especificación de identificador único:}
			
			El identificador único es \emph{CUIT}.			
		
		\subsubsection{Tipo de Componente}
			\paragraph{Definición:} 
			
			Categorías en que son clasificados los Componentes.
			
			\paragraph{Especificación de atributos: \\}
			
			\begin{itemize}
				\item{\emph{Nombre Tipo:}} Nombre del tipo de componente. Por ejemeplo: Placa base, Microprocesador, Memoria RAM, etc.
				\item{\emph{Descripción:}} Breve descripción del tipo de componente.
			\end{itemize}
			
			\paragraph*{Especificación de identificador único:}
			
			El identificador único es \emph{Nombre Tipo}
			
		\subsubsection{Subtipo}
			\paragraph{Definición:} 
			
			Categorías en que son clasificados los Tipos de Componentes.
			
			\paragraph{Especificación de atributos: \\}
			
			\begin{itemize}
				\item{\emph{Nombre Subtipo:}} Nombre del subtipo de compomente. Por ejemplo: Socket 1156 Mini ATX Video OnBoard, Intel i3, etc.
				\item{\emph{Descripcion:}} Breve descripción del subtipo de componente.
				\item{\emph{Stock Mínimo:}} Mínima existencia de este subtipo de componentes que se desea tener en el almacen sin emitir un aviso de stock bajo.
				
			\end{itemize}
			
			\paragraph*{Especificación de identificador único:}
			
			El identificador único es \emph{Nombre}.
