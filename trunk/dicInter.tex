		\subsubsection{Reserva para}
			\paragraph{Definición:} 
				
				Reserva de un Componente hecha por un Operario.
				
				\paragraph{Especificación de atributos: \\}
				
				\begin{itemize}
					\item{\emph{ Número de Serie:}} Número que identifica univocamente a un componente.
					\item{\emph{DNI:}} Documento para la identificación de una persona.
				\end{itemize}
				
				\paragraph*{Especificación de identificador único:}
				
				El identificador único es \emph{Número de Serie+DNI}.
